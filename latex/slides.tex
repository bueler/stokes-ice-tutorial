\section{stage 0 \hspace{9.3mm} the glacier dynamics problem}

\begin{frame}{glacier dynamics using the Stokes model: the high-level view}

\begin{itemize}
\item we apply conservation principles to glaciers:
    \begin{itemize}
    \item[$\circ$] mass conservation \hfill \cmark
    \item[$\circ$] momentum conservation \hfill \cmark
    \item[$\circ$] energy conservation \hfill (not here)
    \end{itemize}
\item incompressibility is one aspect of mass conservation
    \begin{itemize}
    \item[$\circ$] another is the free-surface equation; deferred to \texttt{stage5/}
    \end{itemize}
\item glaciers are a ``slow flow'' problem because the forces of inertia are tiny
    \begin{itemize}
    \item[$\circ$] momentum conservation is called a ``stress balance''
    \item[$\circ$] \alert{velocity \& pressure are instantaneously determined by current geometry \& boundary stresses}
    \end{itemize}

\bigskip
{\scriptsize velocity} \hfill \includegraphics[width=0.8\textwidth]{figs/speed.png}

\medskip
{\scriptsize pressure} \hfill \includegraphics[width=0.8\textwidth]{figs/pressure.png}

\bigskip

\item \alert{Stokes equations $=$ stress balance $+$ incompressibility}
\end{itemize}
\end{frame}


\begin{frame}{Glen-Nye-Stokes equations}

\begin{itemize}
\item notation: $\bu$ is velocity, $p$ is pressure, and $\tau$ is deviatoric stress tensor
\item $\ds D\bu=\tfrac{1}{2}\left(\grad \bu + \grad \bu^\top\right)$ is strain-rate tensor
\item the main equations:
\begin{align*}
- \Div \tau + \nabla p &= \rho \bg &&\text{\emph{stress balance}} &&\phantom{dlfsj fdasj} \\
\Div \bu &= 0 &&\text{\emph{incompressibility}} \\
\tau = B_n &|D\bu|^{(1/n) - 1} D\bu  &&\text{\emph{Glen-Nye flow law}}
\end{align*}

\vspace{-25mm}

%https://www.bl.uk/voices-of-science/interviewees/john-glen#

%https://www.igsoc.org/john-f-nye-1923-2019

\hfill {\tiny John Glen}

\medskip
\hfill \includegraphics[width=0.12\textwidth]{figs/people/jnye.jpg}

\hfill {\tiny John Nye}

\vspace{-10mm}

\item constants (\emph{with default values in demos}):
    \begin{itemize}
    \item[$\circ$] $\rho=910 \,\text{kg}\,\text{m}^{-3}$
    \item[$\circ$] $g=|\bg|=9.81\,\text{m}\,\text{s}^{-2}$
    \item[$\circ$] $n=3$
    \item[$\circ$] $B_n=6.8\times 10^7\,\text{Pa}\,\text{s}^{1/3}$ \hfill {\footnotesize $\leftarrow$ \emph{constant because isothermal}}
    \end{itemize}
\end{itemize}
\end{frame}


\begin{frame}{avoiding unbounded viscosity}

\begin{itemize}
\item the classical viscosity relationship $\tau=2\nu D\bu$ eliminates $\tau$ from the equations, giving a system for $\bu,p$
\item however, the viscosity formula $\nu = \frac{1}{2} B_n |D\bu|^{(1/n) - 1}$ yields unbounded values as $D\bu\to 0$ (vanishing strain rates)
\item so we also regularize with \alert{$\eps>0$} to generate bounded viscosity

\begin{block}{Glen-Nye-Stokes equations with regularized viscosity}

\begin{align*}
- \Div \left(2 \alert{\nu_\eps} D\bu\right) + \nabla p &= \rho \bg &&\text{\emph{stress balance}} \\
\Div \bu &= 0 &&\text{\emph{incompressibility}} \\
\alert{\nu_\eps} = \nu_\eps(|D\bu|) &= \tfrac{1}{2} B_n \left(|D\bu|^2 + \eps\, D_0^2\right)^{((1/n) - 1)/2}
\end{align*}

\smallskip
\end{block}

\item we will use $\eps = 10^{-4}$ and $D_0 = 1\,\text{a}^{-1}$ in the demonstrations
\end{itemize}
\end{frame}


\begin{frame}{boundary conditions, as simple as possible}

\begin{itemize}
\item note that $\sigma = 2 \nu_\eps D\bu - pI$ is the (Cauchy) stress tensor
\item stress-free top:
\begin{equation*}
\sigma \bn = \left(2 \nu_\eps D\bu - pI\right) \bn = \bzero
\end{equation*}

\bigskip
\includegraphics[width=0.9\textwidth]{figs/speed.png}

\bigskip
\item no-slip base:
\begin{equation*}
\bu = \bzero
\end{equation*}
    \begin{itemize}
    \item[$\circ$] \alert{no attempt in these slides to model sliding}
    \end{itemize}
\end{itemize}
\end{frame}


\begin{frame}{finite elements \dots with no details}

\begin{itemize}
\item exact solution of the Stokes model is impossible for most geometries
    \begin{itemize}
    \item[$\circ$] \emph{exercise.}  solve slab-on-a-slope, which is an exception
    \end{itemize}
\item so we solve numerically using the finite element (FE) method
\item the FE method allow us to start from a mesh of triangles {\small (or quadrilaterals, prisms, \dots) over our domain, {\footnotesize then express the approximate solution in terms of functions built from the mesh, {\scriptsize but I will skip the details \dots}}}

\medskip
\input{figs/trihat.tex}

\vspace{-9mm}
\hfill \includegraphics[width=0.7\textwidth]{figs/stage2.png}

\bigskip
\item the whole point of Firedrake is \alert{suppress the details of FE}
    \begin{itemize}
    \item[$\circ$] try Elman, Sylvester \& Wathen (2014) for self-study on FE?
    \end{itemize}
\item however, one needs to know a key step:\, \alert{write a weak form}
\end{itemize}
\end{frame}


\begin{frame}{weak form}

\begin{itemize}
\item start from the equations, also called the \emph{strong form}:
\begin{align}
- \nabla \cdot \left(2 \nu_\eps(|D\bu|)\, D\bu\right) + \nabla p &= \rho \mathbf{g} \\
\Div \bu &= 0
\end{align}
\item multiply (1) by test velocity $\bv$ and (2) by test pressure $q$, integrate by parts, and combine into one nonlinear residual function:
    $$F(\bu,p;\bv,q) = \int_{\Lambda} 2 \nu_\eps(|D\bu|)\, D\bu\,:\,D\bv - p \nabla \cdot \bv - q \nabla \cdot \bu - \rho \mathbf{g} \cdot \bv \,dx$$

    \begin{itemize}
    \item[$\circ$] $\Lambda$ is the domain of ice, and ``$dx$'' $=dx\,dz$ in the planar case
    \end{itemize}

\medskip
\item the actual weak-form problem statement is ``residual $=0$'':
   $$F(\bu,p;\bv,q) = 0 \qquad \text{ for all test functions } \bv \text{ and } q$$
\item the FE method solves this problem, but over a finite-dimensional space
\end{itemize}
\end{frame}


\begin{frame}[fragile]
\frametitle{UFL for the weak form}

\begin{itemize}
\item the nonlinear residual function:
    $$F(\bu,p;\bv,q) = \int_{\Lambda} 2 \nu_\eps(|D\bu|)\, D\bu\,:\,D\bv - p \nabla \cdot \bv - q \nabla \cdot \bu - \rho \mathbf{g} \cdot \bv \,dx$$
with $\nu_\eps(|D\bu|) = \tfrac{1}{2} B_n \left(|D\bu|^2 + \eps\, D_0^2\right)^{((1/n) - 1)/2}$

\smallskip
\item we write this using the \emph{Unified Form Language} (UFL) from Firedrake
\item the residual function translates cleanly:

\medskip
\begin{Verbatim}[fontsize=\small]
def D(w):
    return 0.5 * (grad(w) + grad(w).T)

Du2 = 0.5 * inner(D(u), D(u)) + (eps * Dtyp)**2.0
nu = 0.5 * Bn * Du2**((1.0 / n - 1.0)/2.0)

fbody = rho * Constant((0.0, 0.0, - g))
F = ( inner(2.0 * nu * D(u), D(v)) \
      - p * div(v) - q * div(u) - inner(fbody, v) ) * dx
\end{Verbatim}
\end{itemize}
\end{frame}


\begin{frame}[fragile]
\frametitle{mixed finite elements for fluid problems}

\begin{itemize}
\item the other details we need to know about:
    \begin{itemize}
    \item[{\color{black} 1.}] choose \alert{separate function spaces for the velocity and the pressure}
    \item[{\color{black} 2.}] choose finite element spaces that (the experts say) are \alert{stable}
    \end{itemize}

\bigskip
\item Firedrake makes choosing such mixed elements easy:

\medskip
\begin{Verbatim}[fontsize=\small]
V = VectorFunctionSpace(mesh, 'CG', 2)  # u space is P2
W = FunctionSpace(mesh, 'CG', 1)        # p space is P1
Z = V * W                               # mixed space
up = Function(Z)
u, p = split(up)
v, q = TestFunctions(Z)
\end{Verbatim}

\medskip
    \begin{itemize}
    \item[$\circ$] this $P_2 \times P_1$ choice is called Taylor-Hood
    \item[$\circ$] other mixed spaces are just as easy
    \end{itemize}
\end{itemize}
\end{frame}

\section{\texttt{stage1/} \qquad linear Stokes as a first demo}


\begin{frame}{the linear Stokes equations}

\begin{columns}

\begin{column}{0.6\textwidth}
\begin{itemize}
\item if we make viscosity constant ($\nu_0>0$) then we get a linear Stokes system:
\begin{align*}
- \nabla \cdot \left(2 \nu_0\, D\bu\right) + \nabla p &= \rho \mathbf{g}  \\
\Div \bu &= 0
\end{align*}
\item classical reformat with $2 \nabla \cdot D\bu = \nabla^2 \bu$:
\begin{align*}
- \nu_0 \nabla^2 \bu + \nabla p &= \rho \mathbf{g}  \\
-\Div \bu \qquad \,\,\,\, &= 0
\end{align*}
\item which has symmetric block structure:
  $$\begin{bmatrix} A & B^\top \\ B & 0 \end{bmatrix} \begin{bmatrix} \bu \\ p  \end{bmatrix} = \begin{bmatrix} \bbf \\ 0 \end{bmatrix}$$
since $(-\Div)^\top = \grad$, conceptually-speaking
\end{itemize}
\end{column}

\section{\texttt{stage2/} \qquad Glen-Nye-Stokes for glacier ice}

\begin{frame}{Newton's method}

\begin{columns}
\begin{column}{0.8\textwidth}
\begin{itemize}
\item linear Stokes \small (\texttt{stage1/}) \normalsize needs only a single linear system:
  $$\begin{bmatrix} A & B^\top \\ B & 0 \end{bmatrix} \begin{bmatrix} \bu \\ p  \end{bmatrix} = \begin{bmatrix} \bbf \\ 0 \end{bmatrix}$$
    \begin{itemize}
    \item[$\circ$] default solver: Firedrake $\to$ PETSc KSP $\to$ MUMPS
    \end{itemize}
\item but for Glen-Stokes the residual function is nonlinear in $\bu$:
\small
$$F(\bu,p;\bv,q) = \int_{\Lambda} 2\, \alert{\nu_\eps(|D\bu|)\, D\bu}\,:\,D\bv - p \nabla \cdot \bv - q \nabla \cdot \bu - \rho \mathbf{g} \cdot \bv \,dx$$
\normalsize
\item Firedrake applies Newton's method by default if we write the problem as\, \texttt{solve(F == 0,\dots)}
\item what you need to know about Newton's iteration:
    \begin{itemize}
    \item[$\circ$] it does linearization around each iterate
    \item[$\circ$] Firedrake uses UFL symbolic differentiation for linearization
    \item[$\circ$] Firedrake asks PETSc's SNES to do the Newton iteration
    \item[$\circ$] SNES options will monitor and control the iteration
    \end{itemize}
\end{itemize}
\end{column}

\section{\texttt{stage5/} \qquad evolving glaciers: coupling the free-surface equation}

\begin{frame}{but don't glaciers actually change with the climate?}

\begin{itemize}
\item we do not yet have a ``real'' glacier model
    \begin{itemize}
    \item[$\circ$] the climate (e.g.~surface mass balance) has not appeared as an input
    \item[$\circ$] time derivatives have not appeared either
    \end{itemize}

\medskip
\item[Q.] how do glaciers change shape?

\item[A.] via the \alert{free-surface equation} (or kinematical equation):
   $$\frac{\partial s}{\partial t} - \bu|_s \cdot \bn_s = a$$

    \begin{itemize}
    \item[$\circ$] $z=s(t,x,y)$ is the surface elevation of the glacier
    \item[$\circ$] $a(t,x,y)$ is the surface mass balance
    \item[$\circ$] $\bn_s=\left<-\frac{\partial s}{\partial x},-\frac{\partial s}{\partial y},1\right>$ is a normal vector to the surface
    \item[$\circ$] $\bu|_s$ is the surface value (\emph{trace}) of the Stokes solution velocity
    \end{itemize}

\medskip
\item the free-surface equation holds on $[0,T] \times \Omega$
    \begin{itemize}
    \item[$\circ$] $\Omega \subset \RR^d$ is a fixed map-plane region \hfill {\footnotesize \emph{$d=2$ in reality, $d=1$ for planar}}
    \end{itemize}
\item the \emph{mass continuity} equation is an alternative to the free-surface equation
\end{itemize}
\end{frame}


\begin{frame}{coupled system of equations}

\begin{itemize}
\item so we have a \alert{coupled} system of simultaneous equations:
\begin{align*}
\frac{\partial s}{\partial t} - \bu|_s \cdot \bn_s &= a &&\text{\emph{free-surface equation}} \\
- \Div \left(2 \nu_\eps(|D\bu|)\, D\bu\right) + \nabla p &= \rho \bg &&\text{\emph{stress balance}} \\
\Div \bu &= 0 &&\text{\emph{incompressibility}}
\end{align*}
\end{itemize}

\medskip
\begin{columns}
\begin{column}{0.47\textwidth}
\begin{itemize}
\item the coupling has two aspects:
\end{itemize}
\begin{enumerate}
\item[I.] the free-surface equation has $\bu|_s$ as a coefficient
\item[II.] $s$ determines the evolving domain $\Lambda(t)$ on which the Stokes problem is solved
\end{enumerate}
\end{column}
\begin{column}{0.53\textwidth}
\quad \includegraphics[width=\textwidth]{figs/domain.png}
\end{column}
\end{columns}
\end{frame}


\begin{frame}{weak form of coupled system of equations}

\begin{itemize}
\item get the weak form of the coupled system in the usual way, namely multiply by test functions and integrate by parts:
{\small
\begin{align*}
\int_\Omega \left(\frac{\partial s}{\partial t} - \bu|_s \cdot \bn_s - a\right)\, r\,dx &=0  &&\forall r \\
\int_{\Lambda(t)} 2\, \nu_\eps(|D\bu|)\, D\bu\,:\,D\bv - p \nabla \cdot \bv - \rho \mathbf{g} \cdot \bv\,dx &= 0 &&\forall \bv \\
\int_{\Lambda(t)} - (\Div \bu) q\,dx &= 0 &&\forall q
\end{align*}
}
\end{itemize}

\begin{columns}
\begin{column}{0.55\textwidth}
\begin{itemize}
\item we need to make high-level choices about this coupled system:
    \begin{enumerate}
    \item[1.] time-stepping: explicit or implicit?
    \item[2.] solve it together (\emph{monolithic}) or split?
    \item[3.] spaces and elements for $s,\bu,p$?
    \end{enumerate}
\end{itemize}
\end{column}
\begin{column}{0.45\textwidth}
\phantom{foo}

\vspace{2mm}
\hfill \includegraphics[width=0.9\textwidth]{figs/domain.png}
\end{column}
\end{columns}
\end{frame}


\begin{frame}{challenges in solving the coupled system}

\begin{itemize}
\item early/naive approaches to solving this coupled problem have been fragile
    \begin{itemize}
    \item[$\circ$] the method everyone tries first is to solve Stokes for the current geometry, then advance the surface, and repeat
    \item[$\circ$] this is a \emph{split} and \emph{explicit} approach to the coupled problem
    \item[$\circ$] the result is \alert{limited performance because of poorly-understood time-step restrictions}
    \end{itemize}

\bigskip
\item what are the challenges in designing better schemes?  here are two
\end{itemize}

\begin{enumerate}
\item Recall that $s$ determines the evolving domain $\Lambda(t)$ on which the Stokes problem is solved.  Therefore each time the surface $s$ changes the FE mesh used for the $\bu,p$ solver also changes.  This makes a monolithic or implicit implementations difficult.
\item The coupled problem mixes advective and diffusive aspects to a degree that depends on the particular boundary conditions.
    \begin{itemize}
    \item the precise time-step restriction, and thus the expected performance scaling, remains unclear; see Bueler (2023), and references therein
    \end{itemize}
\end{enumerate}
\end{frame}


\begin{frame}{making choices}

\begin{itemize}
\item we will demonstrate a solver which is \alert{semi-implicit}, \alert{split}, and \alert{stabilized}
    \begin{itemize}
    \item[$\circ$] warning: ``stabilize'' means several things here
    \end{itemize}

\medskip
\item it is based on these choices:
    \begin{enumerate}
    \item[1.] we use first-order, and mostly explicit, time-stepping
        \begin{itemize}
        \item[$\circ$] we add an implicit free-surface \alert{stabilization}
        \item[$\circ$] we determine the time step by a CFL condition
        \end{itemize}
    \item[2.] we split the coupled system, alternating free-surface and Stokes solves
        \begin{itemize}
        \item[$\circ$] we add a \alert{stabilizing} semi-implicit load heuristic to the Stokes solver
        \end{itemize}
    \item[3.] we use $P_1$ elements for $s$, and \alert{stable} $P_2\times P_1$ elements (\emph{as before}) for $\bu,p$
        \begin{itemize}
        \item[$\circ$] note that $s\in P_1$ means $\Lambda(t)$ is polygonal
        \end{itemize}
    \end{enumerate}

\medskip
\item the next slides give more detail on techniques \dots
\end{itemize}
\end{frame}


\newcommand{\jump}[1]{\llbracket #1 \rrbracket}


\begin{frame}{technique 1: implicit edge stabilization in free-surface equation}

\begin{itemize}
\item this technique is from Tominec et al (2026)
    \begin{itemize}
    \item[$\circ$] it is motivated by similar stabilizations used in FE advection solvers
    \end{itemize}
\item add FE \alert{edge stabilization}, applied \alert{implicitly}:
    $$\int_{\Omega^h} \left(\frac{\alert{s_{k+1}^h} - s_k^h}{\Delta t} - \bu|_{s_k^h} \cdot \bn_{s_k^h} - a\right) r^h\,dx + J(\alert{s_{k+1}^h},r^h) = 0$$
where
    $$J(s,r) = \sum_{e\in E^h} \int_e \tfrac{1}{2} h_e^2 \left|\bu|_{s_k^h}\right| \jump{\grad s} \jump{\grad r}\,dS$$

    \begin{itemize}
    \item[$\circ$] $E^h$ is the set of interior edges in the triangulation $\Omega^h$
    \item[$\circ$] $\jump{\bv} = \bv_+ \cdot \bn_+ - \bv_- \cdot \bn_-$ is the jump of a vector field along an edge
    \item[$\circ$] $h_e$ is the average of the diameters of the two cells which meet $e$
    \end{itemize}

\medskip
\item surface velocity and slope are evaluated explicitly, at $t_k$
\item the system to solve for $\alert{s_{k+1}^h}$ is \emph{linear}
\end{itemize}
\end{frame}


\begin{frame}{technique 2: semi-implicit coupling heuristic in stress balance}

\begin{itemize}
\item also from Tominec et al (2026), but based on L\"ofgren et al (2022)
\item the stress balance is modified to stabilize the coupling
\item we \alert{add an extrapolated (and symmetrized) load on the upper surface}:
{\small
\begin{align*}
\int_{\Lambda(t_k)} 2\, \nu_\eps(|D\bu|)\, D\bu\,:\,D\bv - p \nabla \cdot \bv - \rho \mathbf{g} \cdot \bv\,dx \qquad& \\
\alert{+ \int_{\Gamma_s(t_k)} \rho g \Delta t \left[\frac{1}{2} (\bu\cdot \bn_{s_k}) + a\right] (\bv\cdot \bn) \,ds} &= 0
\end{align*}

    \begin{itemize}
    \item[$\circ$] the Stokes domain $\Lambda(t_k)$ is defined by the current surface elevation $s_k$
    \item[$\circ$] $\Gamma_s(t_k)$ is the upper surface of $\Lambda(t_k)$
    \item[$\circ$] $\bn_{s_k}=\left<-\grad s_k,1\right>$ is an outward normal on $\Gamma_s(t_k)$
    \item[$\circ$] $\bn=\bn_{s_k}/|\bn_{s_k}|$ is the outward \emph{unit} normal 
    \end{itemize}
}
\item L\"ofgren et al (2022) use Reynold's transport theorem to motivate this type of stabilization of the coupled system
\item Tominec et al (2026) \alert{prove} that techniques 1 \& 2 together stabilize the $L^2$ norm of the evolving surface
\end{itemize}
\end{frame}

\begin{frame}{glaciers have margins}

\begin{itemize}
\item however, the \alert{surface elevation must be above the bed}
\item enforcing this obvious fact provides the (lateral) free-boundary condition for a land-based glacier
\item solving the coupled problem subject to this condition determines the location of the \alert{glacier margin}
    \begin{itemize}
    \item[$\circ$] the thickness and flux are both zero at the margin
    \end{itemize}
\end{itemize}

\begin{frame}{variational inequality for free surface}

\begin{itemize}
\item suppose a (time-independent) bed elevation function $b(x,y)$
    \begin{itemize}
    \item[$\circ$] the meshed $\text{CG}_1$ bed is denoted $b^h$
    \end{itemize}
\item define the FE \alert{admissible set $\cK^h = \left\{r^h \in \text{CG}_1 \,:\, r^h\ge b^h\right\}$}
\item the weak form solved by the FE free surface \alert{$s_{k+1}^h \in \cK^h$} is a \alert{variational inequality}:
    $$\int_\Omega \left(\frac{s_{k+1}^h - s_k^h}{\Delta t} - \bu|_{s_k^h} \cdot \bn_{s_k^h} - a\right)\, (r^h-s_{k+1}^h)\,dx + J(s_{k+1}^h,r^h-s_{k+1}^h) \ge 0$$
for all $r^h$ in $\cK^h$

    \begin{itemize}
    \item[$\circ$] this includes the edge

stabilization
    \end{itemize}

\item the computed margin

location is where

$s_{k+1}^h$ meets $b^h$

\end{itemize}
