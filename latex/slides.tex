\section{stage 0 \hspace{9.3mm} the glacier dynamics problem}

\begin{frame}{glacier dynamics using Stokes}

\begin{itemize}
\item at a high level, one applies these conservation principles to glaciers and ice sheets:
    \begin{itemize}
    \item[$\circ$] mass conservation \hfill \cmark
    \item[$\circ$] momentum conservation \hfill \cmark
    \item[$\circ$] energy conservation \hfill (not here)
    \end{itemize}
\item incompressibility is one aspect of mass conservation
    \begin{itemize}
    \item[$\circ$] another aspect is the free-surface equation; deferred to \texttt{stage5/}
    \end{itemize}
\item because glaciers are a slow flow problem, \alert{the Stokes model determines velocity and pressure from current geometry and boundary stresses}
    \begin{itemize}
    \item[$\circ$] thus momentum conservation is called a ``stress balance''
    \end{itemize}

\bigskip
{\scriptsize velocity} \hfill \includegraphics[width=0.8\textwidth]{figs/speed.png}

\medskip
{\scriptsize pressure} \hfill \includegraphics[width=0.8\textwidth]{figs/pressure.png}

\bigskip

\item Stokes equations $=$ stress balance $+$ incompressibility
\end{itemize}
\end{frame}


\begin{frame}{Glen-Stokes equations}

\begin{itemize}
\item notation: $\bu$ is velocity, $p$ is pressure, and $\tau$ is the deviatoric stress tensor
\item also $\ds D\bu=\tfrac{1}{2}\left(\grad \bu + \grad \bu^\top\right)$ is the strain-rate tensor
\item the main equations:

\begin{columns}
\begin{column}{0.8\textwidth}
\begin{align*}
- \Div \tau + \nabla p &= \rhoi \bg &&\text{\emph{stress balance}} \\
\Div \bu &= 0 &&\text{\emph{incompressibility}} \\
\tau = B_n &|D\bu|^{(1/n) - 1} D\bu  &&\text{\emph{Glen-Nye flow law}}
\end{align*}
\end{column}

\begin{column}{0.2\textwidth}

\vspace{3mm}
%https://www.bl.uk/voices-of-science/interviewees/john-glen#
\hfill \includegraphics[width=0.6\textwidth]{figs/people/jglen.png}

\hfill {\tiny John Glen}
\end{column}
\end{columns}

\smallskip

\item constants:
    \begin{itemize}
    \item[$\circ$] $\rhoi=910 \,\text{kg}\,\text{m}^{-3}$
    \item[$\circ$] $g=|\bg|=9.81\,\text{m}\,\text{s}^{-2}$
    \item[$\circ$] $n=3$
    \item[$\circ$] isothermal, so $B_n$ is constant: $B_n=6.8\times 10^7\,\text{Pa}\,\text{s}^{1/3}$
    \end{itemize}
\end{itemize}
\end{frame}


\begin{frame}{avoiding unbounded viscosity}

\begin{itemize}
\item one may use a classical viscosity relationship $\tau=2\nu D\bu$ to eliminate $\tau$ from the equations, to give system for $\bu,p$
\item however, the coefficient $|D\bu|^{(1/n) - 1}$ in the flow law yields unbounded values as $D\bu\to 0$, that is, as strain rates vanish
\item so we also regularize to generate bounded viscosity

\begin{block}{Glen-Stokes equations with regularized viscosity}

\begin{align*}
- \Div \left(2 \nu_\eps(|D\bu|)\, D\bu\right) + \nabla p &= \rhoi \bg &&\text{\emph{stress balance}} \\
\Div \bu &= 0 &&\text{\emph{incompressibility}}
\end{align*}

\smallskip
\end{block}

\item where
\begin{equation*}
\nu_\eps(|D\bu|) = \tfrac{1}{2} B_n \left(|D\bu|^2 + \eps\, D_0^2\right)^{((1/n) - 1)/2}
\end{equation*}
\item we will use $\eps = 10^{-4}$ and $D_0 = 1\,\text{a}^{-1}$
\end{itemize}
\end{frame}


\begin{frame}{boundary conditions, as simple as possible}

\begin{itemize}
\item note that $\sigma = 2 \nu_\eps(|D\bu|) D\bu - pI$ is the (Cauchy) stress tensor
\item stress-free top:
\begin{equation*}
\sigma \bn = \left(2 \nu_\eps(|D\bu|) D\bu - pI\right) \bn = \bzero
\end{equation*}

\bigskip
\includegraphics[width=0.9\textwidth]{figs/speed.png}

\bigskip
\item no-slip base:
\begin{equation*}
\bu = \bzero
\end{equation*}
    \begin{itemize}
    \item[$\circ$] no attempt in these slides to model sliding
    \end{itemize}
\end{itemize}
\end{frame}

\section{\texttt{stage1/} \qquad linear Stokes as a first demo}


\begin{frame}{the linear Stokes equations}

\begin{columns}

\begin{column}{0.6\textwidth}
\begin{itemize}
\item if we make viscosity constant ($\nu_0$) then we get a linear Stokes system
\begin{align*}
- \nabla \cdot \left(2 \nu_0\, D\bu\right) + \nabla p &= \rhoi \mathbf{g}  \\
\Div \bu &= 0
\end{align*}
\item use $2 \nabla \cdot D\bu = \nabla^2 \bu$ to reformat:
\begin{align*}
- \nu_0 \nabla^2 \bu + \nabla p &= \rhoi \mathbf{g}  \\
-\Div \bu \qquad \,\,\,\, &= 0
\end{align*}
\item which has symmetric block structure
  $$\begin{bmatrix} A & B^\top \\ B & 0 \end{bmatrix} \begin{bmatrix} \bu \\ p  \end{bmatrix} = \begin{bmatrix} \bbf \\ 0 \end{bmatrix}$$
since $(-\Div)^\top = \grad$, conceptually-speaking
\end{itemize}
\end{column}

\begin{column}{0.4\textwidth}

\begin{frame}{finite elements \dots with no details}

\begin{itemize}
\item exact solution of the Stokes model is impossible for most geometries
    \begin{itemize}
    \item[$\circ$] \emph{exercise.}  solve slab-on-a-slope (an exception)
    \end{itemize}
\item so we solve numerically using the finite element (FE) method
\item in summary, the FE method allow us to start from a mesh of triangles {\small (or quadrilaterals, prisms, \dots) over our domain, {\footnotesize then express the approximate solution in terms of functions built from the mesh, {\scriptsize but I will skip the details \dots}}}

\medskip
\input{figs/trihat.tex}

\vspace{-9mm}
\hfill \includegraphics[width=0.7\textwidth]{figs/stage2.png}

\bigskip
\item the whole point of Firedrake is \alert{suppress the details of FE}
\item however, one needs to know the first step:\, \alert{write a weak form}
\end{itemize}
\end{frame}


\begin{frame}[fragile]
\frametitle{weak form}

\begin{itemize}
\item start from the equations, also called the \emph{strong form}:
\small
\begin{align}
- \nabla \cdot \left(2 \nu_\eps(|D\bu|)\, D\bu\right) + \nabla p &= \rhoi \mathbf{g} \\
\Div \bu &= 0
\end{align}
\normalsize
\item multiply (1) by test velocity $\bv$ and (2) by test pressure $q$, integrate by parts, and combine into one nonlinear residual function:
    $$F(\bu,p;\bv,q) = \int_{\Omega} 2 \nu_\eps(|D\bu|)\, D\bu\,:\,D\bv - p \nabla \cdot \bv - q \nabla \cdot \bu - \rhoi \mathbf{g} \cdot \bv \,dx$$
\item in Firedrake's language (= \emph{Unified Form Language}) it looks like:

\medskip
\begin{Verbatim}[fontsize=\small]
fbody = rho * Constant((0.0, 0.0, - g))
Du2 = 0.5 * inner(D(u), D(u)) + (eps * Dtyp)**2.0
nu = 0.5 * Bn * Du2**((1.0 / n - 1.0)/2.0)
F = ( inner(2.0 * nu * D(u), D(v)) \
      - p * div(v) - q * div(u) - inner(fbody, v) ) * dx
\end{Verbatim}

\medskip
\item this weak-form problem statement is solved by the FE method:
   $$F(\bu,p;\bv,q) = 0 \qquad \text{ for all } \bv \text{ and } q$$
\end{itemize}
\end{frame}


\begin{frame}[fragile]
\frametitle{mixed finite elements for fluid problems}

\begin{itemize}
\item the other details we need to know about:
    \begin{itemize}
    \item[{\color{black} 1.}] choose \alert{separate function spaces for the velocity and the pressure}
    \item[{\color{black} 2.}] choose spaces that the experts say are \alert{stable}
    \end{itemize}

\bigskip
\item Firedrake makes choosing mixed elements easy:

\medskip
\begin{Verbatim}[fontsize=\small]
V = VectorFunctionSpace(mesh, 'Lagrange', 2)  # u space
W = FunctionSpace(mesh, 'Lagrange', 1)        # p space
Z = V * W                                     # mixed space
up = Function(Z)
u, p = split(up)
v, q = TestFunctions(Z)
\end{Verbatim}

\medskip
    \begin{itemize}
    \item[$\circ$] this $P_2 \times P_1$ choice is called Taylor-Hood
    \item[$\circ$] other mixed spaces are just as easy
    \end{itemize}
\end{itemize}
\end{frame}

\section{\texttt{stage2/} \qquad Glen-Stokes for glacier ice}

\begin{frame}{Newton's method}

\begin{columns}
\begin{column}{0.8\textwidth}
\begin{itemize}
\item linear Stokes \small (\texttt{stage1/}) \normalsize needs only a single linear system:
  $$\begin{bmatrix} A & B^\top \\ B & 0 \end{bmatrix} \begin{bmatrix} \bu \\ p  \end{bmatrix} = \begin{bmatrix} \bbf \\ 0 \end{bmatrix}$$
    \begin{itemize}
    \item[$\circ$] default solver: Firedrake $\to$ PETSc KSP $\to$ MUMPS
    \end{itemize}
\item but for Glen-Stokes the weak form is nonlinear in $\bu$:
\small
$$F(\bu,p;\bv,q) = \int_{\Omega} 2\, \alert{\nu_\eps(|D\bu|)\, D\bu}\,:\,D\bv - p \nabla \cdot \bv - q \nabla \cdot \bu - \rhoi \mathbf{g} \cdot \bv \,dx$$
\normalsize
\item Firedrake automatically applies Newton's method if we write the problem as \texttt{solve(F == 0,\dots)}
\item what you need to know about Newton's iteration:
    \begin{itemize}
    \item[$\circ$] it does linearization around each iterate
    \item[$\circ$] Firedrake uses UFL symbolic differentiation for linearization
    \item[$\circ$] Firedrake asks PETSc's SNES to do the Newton iteration
    \item[$\circ$] SNES options will monitor and control the Newton iteration
    \end{itemize}
\end{itemize}
\end{column}

\section{\texttt{stage5/} \qquad evolving glaciers: coupling the free-surface equation}

\begin{frame}{but don't glaciers actually change with the climate?}

\begin{itemize}
\item we do not yet have a ``real'' glacier model
    \begin{itemize}
    \item[$\circ$] the climate (e.g.~surface mass balance) has not appeared as an input
    \item[$\circ$] time derivatives have not appeared either
    \end{itemize}

\medskip
\item[Q.] how do glaciers change shape?

\item[A.] via the \alert{free-surface equation} (or kinematical equation):
   $$\frac{\partial s}{\partial t} - \bu|_s \cdot \bn_s = a$$

    \begin{itemize}
    \item[$\circ$] $z=s(t,x,y)$ is the \alert{surface elevation} of the glacier
    \item[$\circ$] $a(t,x,y)$ is the \alert{surface mass balance}
    \item[$\circ$] $\bn_s=\left<-\frac{\partial s}{\partial x},-\frac{\partial s}{\partial y},1\right>$ is a \alert{normal vector} to the surface
    \item[$\circ$] $\bu_s$ is the surface value of the Stokes solution velocity
    \end{itemize}

\medskip
\item the free-surface equation holds on $[0,T] \times \Omega$
    \begin{itemize}
    \item[$\circ$] $\Omega \subset \RR^d$ is a fixed map-plane region, with $d=2$ in reality, but $d=1$ in the coming demo
    \end{itemize}
\end{itemize}
\end{frame}


\begin{frame}{coupled system of equations}

\begin{itemize}
\item the strong form of the \alert{coupled} system:
\begin{align*}
\frac{\partial s}{\partial t} - \bu|_s \cdot \bn_s &= a &&\text{\emph{free-surface equation}} \\
- \Div \left(2 \nu_\eps(|D\bu|)\, D\bu\right) + \nabla p &= \rhoi \bg &&\text{\emph{stress balance}} \\
\Div \bu &= 0 &&\text{\emph{incompressibility}}
\end{align*}
\end{itemize}

\begin{columns}
\begin{column}{0.47\textwidth}
\begin{itemize}
\item the free-surface equation is coupled to the Stokes problem
    \begin{itemize}
    \item[$\circ$] \alert{free-surface equation has $\bu|_s$ as a coefficient}
    \end{itemize}
\item the Stokes problem is coupled to the surface elevation
    \begin{itemize}
    \item[$\circ$] \alert{$s$ determines the evolving domain $\Lambda(t)$ on which the Stokes problem is solved}
    \end{itemize}
\end{itemize}
