\section{\emph{theory} \hspace{11mm} equations for the glacier dynamics problem}

\begin{frame}{glacier dynamics using a Stokes model: the high-level view}

\begin{itemize}
\item we apply conservation principles to glaciers:
    \begin{itemize}
    \item[$\circ$] mass conservation \hfill \cmark
    \item[$\circ$] momentum conservation \hfill \cmark
    \item[$\circ$] energy conservation \hfill (not here)
    \end{itemize}
\item incompressibility is one aspect of mass conservation
    \begin{itemize}
    \item[$\circ$] another is the free-surface equation \dots see \texttt{stage4/}
    \end{itemize}
\item glaciers are a ``slow flow'' problem because inertial forces are tiny
    \begin{itemize}
    \item[$\circ$] momentum conservation is therefore called a ``stress balance''
    \item[$\circ$] velocity \& pressure are \alert{instantaneously determined} by glacier geometry and boundary stresses
    \end{itemize}
\item \alert{Stokes equations $=$ stress balance $+$ incompressibility}
\end{itemize}

\bigskip\bigskip

{\scriptsize velocity} \hfill \includegraphics[width=0.8\textwidth]{figs/speed.png}

\medskip
{\scriptsize pressure} \hfill \includegraphics[width=0.8\textwidth]{figs/pressure.png}
\end{frame}


\begin{frame}{Glen-Nye-Stokes equations}

\begin{itemize}
\item the canonical glacier dynamics equations are from Glen, Nye, and Stokes
\item the main equations:
\begin{align*}
- \Div \tau + \nabla p &= \rho \bg &&\text{\emph{stress balance}} &&\phantom{dlfsj fdasj} \\
\Div \bu &= 0 &&\text{\emph{incompressibility}} \\
\tau = B_n &|D\bu|^{(1/n) - 1} D\bu  &&\text{\emph{Glen-Nye flow law}}
\end{align*}

    \begin{itemize}
    \item[$\circ$] $\bu$ is velocity, $p$ is pressure, $\tau$ is the deviatoric stress tensor,

and $\ds D\bu=\tfrac{1}{2}\left(\grad \bu + \grad \bu^\top\right)$ is the strain-rate tensor
    \item[$\circ$] our model has \emph{constant ice density}; incompressibility follows
    \end{itemize}

\medskip

\hfill {\tiny John Nye}

\vspace{-10mm}

\item constants (\emph{with default values in demos}):
    \begin{itemize}
    \item[$\circ$] $\rho=910 \,\text{kg}\,\text{m}^{-3}$
    \item[$\circ$] $|\bg|=9.81\,\text{m}\,\text{s}^{-2}$
    \item[$\circ$] $n=3$
    \item[$\circ$] $B_n=6.8\times 10^7\,\text{Pa}\,\text{s}^{1/3}$ \hfill {\footnotesize $\leftarrow$ \emph{constant because isothermal}}
    \end{itemize}
\end{itemize}
\end{frame}


\begin{frame}{viscosity, and avoiding unbounded values}

\begin{itemize}
\item the classical viscosity relationship $\tau=2\nu D\bu$ eliminates $\tau$ from the equations, giving a system for $\bu,p$
\item however, the unmodified Glen-Nye viscosity formula yields unbounded values as $D\bu\to 0$, i.e.~for vanishing strain rates
    \begin{itemize}
    \item[$\circ$] note that ``$(1/n) - 1$'' in $\nu = \frac{1}{2} B_n |D\bu|^{(1/n) - 1}$ is a negative power
    \end{itemize}
\item so we also regularize with \alert{small $\eps>0$} to generate bounded viscosity

\begin{block}{Glen-Nye-Stokes equations with regularized viscosity}

\begin{align*}
- \Div \left(2 \alert{\nu_\eps(|D\bu|)} D\bu\right) + \nabla p &= \rho \bg &&\text{\emph{stress balance}} \\
\phantom{\int} \Div \bu &= 0 &&\text{\emph{incompressibility}} \\
\alert{\nu_\eps(|D\bu|)} &= \tfrac{1}{2} B_n \left(|D\bu|^2 + (\eps\, D_0)^2\right)^{((1/n) - 1)/2} &&\text{\emph{flow law}}
\end{align*}

\smallskip
\end{block}

    \begin{itemize}
    \item[$\circ$] we use $\eps = 10^{-4}$ and $D_0 = 1\,\text{a}^{-1}$ in the demonstrations
    \end{itemize}
\end{itemize}
\end{frame}


\begin{frame}{boundary conditions, as simple as possible}

\begin{itemize}
\item note that $\sigma = 2 \nu_\eps D\bu - pI$ is the \alert{Cauchy stress tensor}
\item stress-free top:
\begin{equation*}
\sigma \bn = \left(2 \nu_\eps D\bu - pI\right) \bn = \bzero
\end{equation*}

    \begin{itemize}
    \item[$\circ$] atmospheric pressure could be applied at surface \dots no interesting effect
    \end{itemize}
\item no-slip base:
\begin{equation*}
\bu = \bzero
\end{equation*}

    \begin{itemize}
    \item[$\circ$] \alert{no attempt to model basal sliding} in this tutorial
    \end{itemize}
\end{itemize}

\begin{frame}{finite elements \dots with no details}

\begin{itemize}
\item by-hand solution of the Stokes model is impossible for most geometries
    \begin{itemize}
    \item[$\circ$] \emph{exercise.}  solve slab-on-a-slope, which is an exception
    \end{itemize}
\item so we solve numerically using the finite element (FE) method
\item the FE method allow us to start from a mesh of triangles {\small (or quadrilaterals, prisms, \dots) over our domain, {\footnotesize then express the approximate solution in terms of functions built from the mesh, {\scriptsize but I will skip the details \dots}}}

\medskip

\bigskip
\item the whole point of Firedrake is \alert{suppress the details} of FE!
    \begin{itemize}
    \item[$\circ$] but try Elman, Sylvester \& Wathen (2014) for self-study?
    \end{itemize}
\item however, one needs to know a key step:\, \alert{write a weak form}
\end{itemize}
\end{frame}


\begin{frame}{the weak form}

\begin{itemize}
\item start from the equations, now called the \emph{strong form}:
\begin{align}
- \nabla \cdot \left(2 \nu_\eps(|D\bu|)\, D\bu\right) + \nabla p &= \rho \mathbf{g} \\
\Div \bu &= 0
\end{align}
\item multiply (1) by test velocity $\bv$ and (2) by test pressure $q$, integrate by parts, and combine into one \alert{nonlinear residual function}:
    $$F(\bu,p;\bv,q) = \int_{\Lambda} 2 \nu_\eps(|D\bu|)\, D\bu\,:\,D\bv - p \nabla \cdot \bv - q \nabla \cdot \bu - \rho \mathbf{g} \cdot \bv \,dx$$

    \begin{itemize}
    \item[$\circ$] $\Lambda$ is the spatial domain of ice
    \item[$\circ$] ``$dx$'' $=dx\,dy\,dz$ in 3D, and ``$dx$'' $=dx\,dz$ in the planar case
    \end{itemize}

\medskip
\item the actual \alert{weak-form problem statement} is ``residual $=0$'':
   $$F(\bu,p;\bv,q) = 0 \qquad \text{ for all test functions } \bv \text{ and } q$$

    \begin{itemize}
    \item[$\circ$] the FE method solves the same problem, but over a finite-dimensional space
    \end{itemize}
\end{itemize}
\end{frame}


\begin{frame}[fragile]
\frametitle{UFL for the weak form}

\begin{itemize}
\item nonlinear residual function from last slide:
    $$F(\bu,p;\bv,q) = \int_{\Lambda} 2 \nu_\eps(|D\bu|)\, D\bu\,:\,D\bv - p \nabla \cdot \bv - q \nabla \cdot \bu - \rho \mathbf{g} \cdot \bv \,dx$$
with $\nu_\eps(|D\bu|) = \tfrac{1}{2} B_n \left(|D\bu|^2 + (\eps\, D_0)^2\right)^{((1/n) - 1)/2}$

\smallskip
\item we write this using the \emph{Unified Form Language} (UFL) in Firedrake:

\medskip
\begin{Verbatim}[fontsize=\small]
def D(w):
    return 0.5 * (grad(w) + grad(w).T)

Du2 = 0.5 * inner(D(u), D(u)) + (eps * Dtyp)**2.0
nu = 0.5 * Bn * Du2**((1.0 / n - 1.0)/2.0)

fbody = rho * Constant((0.0, 0.0, - g))
F = ( 2.0 * nu * inner(D(u), D(v)) \
      - p * div(v) - q * div(u) - inner(fbody, v) ) * dx
\end{Verbatim}

\medskip
    \begin{itemize}
    \item[$\circ$] \alert{Python UFL code is quite similar to the math}
    \end{itemize}
\end{itemize}
\end{frame}


\begin{frame}[fragile]
\frametitle{mixed finite elements for fluid problems}

\begin{itemize}
\item other details we need to know about:
    \begin{itemize}
    \item[{\color{black} 1.}] choose \alert{separate function spaces} for the velocity and the pressure
    \item[{\color{black} 2.}] choose finite element spaces that (the experts say) are \alert{stable}
    \end{itemize}

\bigskip
\item Firedrake makes choosing such \alert{mixed elements} easy:

\medskip
\begin{Verbatim}[fontsize=\small]
V = VectorFunctionSpace(mesh, 'CG', 2)  # u space is P2
W = FunctionSpace(mesh, 'CG', 1)        # p space is P1
Z = V * W                               # mixed space
up = Function(Z)
u, p = split(up)
v, q = TestFunctions(Z)
\end{Verbatim}

\medskip
    \begin{itemize}
    \item[$\circ$] this $P_2 \times P_1$ choice is called Taylor-Hood
    \item[$\circ$] other mixed spaces are just as easy
    \item[$\circ$] we choose $P_2\times DG_0$ in \texttt{stage4/}, for reasons \dots
    \end{itemize}
\end{itemize}
\end{frame}

\section{\texttt{stage1/} \qquad first demo: linear Stokes}


\begin{frame}{the linear Stokes equations}

\begin{columns}

\begin{column}{0.6\textwidth}
\begin{itemize}
\item if we make viscosity constant ($\nu_0>0$) then we get a linear Stokes system:
\begin{align*}
- \nabla \cdot \left(2 \nu_0\, D\bu\right) + \nabla p &= \rho \mathbf{g}  \\
\Div \bu &= 0
\end{align*}
\item classical reformat with $2 \nabla \cdot D\bu = \nabla^2 \bu$:
\begin{align*}
- \nu_0 \nabla^2 \bu + \nabla p &= \rho \mathbf{g}  \\
-\Div \bu \qquad \,\,\,\, &= 0
\end{align*}
\item which has symmetric block structure:
  $$\begin{bmatrix} A & B^\top \\ B & 0 \end{bmatrix} \begin{bmatrix} \bu \\ p  \end{bmatrix} = \begin{bmatrix} \bbf \\ 0 \end{bmatrix}$$

    \begin{itemize}
    \item[$\circ$] $A=-\nu_0 \nabla^2 \ge 0$
    \item[$\circ$] $B^\top=(-\Div)^\top = \grad$ (integration-by-parts)
    \end{itemize}
\end{itemize}
\end{column}

\section{\texttt{stage2/} \qquad Glen-Nye-Stokes for glacier ice}

\begin{frame}{Newton's method}

\begin{columns}
\begin{column}{0.8\textwidth}
\begin{itemize}
\item linear Stokes in \texttt{stage1/} solves a single linear system:
  $$\begin{bmatrix} A & B^\top \\ B & 0 \end{bmatrix} \begin{bmatrix} \bu \\ p  \end{bmatrix} = \begin{bmatrix} \bbf \\ 0 \end{bmatrix}$$

    \begin{itemize}
    \item[$\circ$] default solver: Firedrake $\to$ PETSc KSP $\to$ MUMPS
    \end{itemize}
\item but Glen-Nye-Stokes residual function is \alert{nonlinear in $\bu$}:
\small
$$F(\bu,p;\bv,q) = \int_{\Lambda} 2\, \alert{\nu_\eps(|D\bu|)\, D\bu}\,:\,D\bv - p \nabla \cdot \bv - q \nabla \cdot \bu - \rho \mathbf{g} \cdot \bv \,dx$$
\normalsize
\item fortunately, Firedrake applies Newton's method by default if we write the problem as\, \texttt{solve(F == 0,\dots)}
\item what you need to know about Newton's iteration:
    \begin{itemize}
    \item[$\circ$] it does linearization around each iterate
    \item[$\circ$] Firedrake uses UFL symbolic differentiation for linearization
    \item[$\circ$] Firedrake asks PETSc's SNES to do the Newton iteration
    \item[$\circ$] SNES options will monitor and control the iteration
    \end{itemize}
\end{itemize}
\end{column}

\section{\emph{theory} \hspace{11mm} evolving glaciers: free-surface equation coupled to Stokes}

\begin{frame}{but don't glaciers actually change with the climate?}

\begin{itemize}
\item we do not yet have a evolving glacier model
    \begin{itemize}
    \item[$\circ$] time derivatives have not appeared!
    \item[$\circ$] the climate (e.g.~surface mass balance) has not appeared as an input
    \end{itemize}

\medskip
\item[Q.] how do glaciers change shape?

\item[A.] via the \alert{free-surface equation}, a kinematical and mass conservation statement:
   $$\frac{\partial s}{\partial t} - \bu|_s \cdot \bn_s = a$$

    \begin{itemize}
    \item[$\circ$] $z=s(t,x,y)$ is the surface elevation of the glacier

\smallskip
    \item[$\circ$] $\bu|_s\strut$ is the surface value (\emph{trace}) of the ice flow velocity
    \item[$\circ$] $\bn_s=\left<-\frac{\partial s}{\partial x},-\frac{\partial s}{\partial y},1\right>$ is a normal vector to the surface
    \item[$\circ$] $a(t,x,y)$ is the surface mass balance, the source term
    \end{itemize}

\medskip
\item equation holds on $(0,T) \times \Omega$, where $\Omega \subset \RR^d\strut$ is a fixed map-plane region
\item initial surface $s(0,x,y)$ is needed
\end{itemize}
\end{frame}


\begin{frame}{coupled system of equations}

\begin{itemize}
\item so we have a \alert{coupled} system of simultaneous equations

\begin{block}{coupled free-surface \& Glen-Nye-Stokes equations}
\vspace{-2mm}

\begin{align*}
\frac{\partial s}{\partial t} - \bu|_s \cdot \bn_s &= a &&\text{\emph{free-surface equation}} \\
- \Div \left(2 \nu_\eps\, D\bu\right) + \nabla p &= \rho \bg &&\text{\emph{stress balance}} \strut \\
\Div \bu &= 0 &&\text{\emph{incompressibility}} \phantom{\int}
\end{align*}
\end{block}
\end{itemize}

\medskip
\begin{columns}
\begin{column}{0.5\textwidth}
\begin{itemize}
\item 2 coupling directions:
\end{itemize}
\begin{enumerate}
\item[I.] the free-surface equation has $\bu|_s$ as a coefficient
\item[II.] $s(t,x,y)$ determines the \alert{evolving domain $\Lambda(t)$} on which the Stokes problem is solved at each instant
\end{enumerate}
\end{column}
\begin{column}{0.5\textwidth}
\mbox{\, \includegraphics[width=0.95\textwidth]{figs/domain.png}}
\end{column}
\end{columns}
\end{frame}


\begin{frame}{weak form of coupled system of equations}

\begin{itemize}
\item we get the weak form of the coupled system in the usual way \dots

multiply by test functions and integrate by parts:
{\small
\begin{align*}
\int_\Omega \left(\frac{\partial s}{\partial t} - \bu|_s \cdot \bn_s - a\right)\, r\,dx &=0  &&\forall r \\
\int_{\Lambda(t)} 2\, \nu_\eps(|D\bu|)\, D\bu\,:\,D\bv - p \nabla \cdot \bv - \rho \mathbf{g} \cdot \bv\,dx &= 0 &&\forall \bv \\
\int_{\Lambda(t)} - (\Div \bu) q\,dx &= 0 &&\forall q
\end{align*}
}
\end{itemize}

\begin{columns}
\begin{column}{0.55\textwidth}
\begin{itemize}
\item high-level choices to make about the coupled system:
    \begin{enumerate}
    \item[1.] time-stepping: explicit or implicit?
    \item[2.] solved together (\emph{monolithic}) or split?
    \item[3.] spaces and elements for $s,\bu,p$?
    \end{enumerate}
\end{itemize}
\end{column}
\begin{column}{0.45\textwidth}
\phantom{foo}

\end{column}
\end{columns}
\end{frame}


\begin{frame}{challenges in solving the coupled system}

\begin{itemize}
\item early/naive approaches to solving this coupled problem have been fragile
\item the method everyone tries first is to solve Glen-Nye-Stokes for the current geometry, then advance the surface, and repeat
    \begin{itemize}
    \item[$\circ$] This is an \emph{explicit} approach on \emph{splitting} the coupling.
    \item[$\circ$] Resulting performance is \alert{stability limited}, but in a manner which is not precisely understood.
    \end{itemize}

\medskip
\item<2> what are the underlying challenges in designing better schemes?
\end{itemize}

\begin{enumerate}
\item<2> Because $s$ determines the evolving domain $\Lambda(t)$ on which the Stokes problem is solved, the geometry of the mesh used for the $\bu,p$ solver changes at every time step.  This makes monolithic and/or implicit implementations difficult.  (But see Ahlkrona et al (2026).)
\item<2> The coupled problem mixes advective and diffusive aspects, with proportions depending on the particular boundary conditions.
    \begin{itemize}
    \item Precise time-step restrictions remain unclear.
    \item Expected performance scaling remains unclear (B 2023).
    \item However, \texttt{stage4/} will apply some recent theoretical progress!
    \end{itemize}
\end{enumerate}
\end{frame}


\begin{frame}{also, glaciers have moving margins!}

\begin{itemize}
\item the \alert{surface elevation must be above the bed}
\item enforcing this obvious fact provides the (lateral) free-boundary condition for a land-based glacier
    \begin{itemize}
    \item[$\circ$] the weak form is a \alert{variational inequality} (B 2021b, Chapter 12)
    \item[$\circ$] the thickness and flux are both zero at the free boundary
    \end{itemize}
\item solving the coupled problem subject to this condition determines the location of the moving \alert{glacier margin} ($=$ free boundary)

\end{center}
\end{itemize}
\end{frame}

\begin{frame}{new techniques for \texttt{stage4/}}

\begin{itemize}
\item our solver is based on three \alert{new techniques} which modify the weak form of the coupled problem:
\end{itemize}

\begin{enumerate}
\item[1.] first-order, and mostly explicit, time-stepping
        \begin{itemize}
        \item[A.] but we add an implicit \alert{edge stabilization} in the free-surface equation
        \item[B.] and we determine the time step by a CFL condition
        \end{itemize}

\item[2.] alternating free-surface and Stokes solves
        \begin{itemize}
        \item[$\circ$] but we add a \alert{load stabilization} coupling to the Stokes solver
        \end{itemize}

\item[3.] the coupled free-surface problem is solved as a \alert{variational inequality} which determines the glacier extent and margin
        \begin{itemize}
        \item[A.] we apply a bounds-respecting Newton method
        \item[B.] we allow variable ice extent on the extruded mesh
        \item[C.] we prefer pressure elements with interior degrees of freedom

\newcommand{\jump}[1]{\llbracket #1 \rrbracket}


\begin{frame}{technique 1A: implicit edge stabilization in free-surface equation}

\begin{itemize}
\item this technique is from Tominec et al (2026)
    \begin{itemize}
    \item[$\circ$] see also Burman \& Hansbro (2004), Ern \& Guermond (2013)
    \end{itemize}
\item add FE \alert{edge stabilization}, applied semi-\alert{implicitly}:
    $$\int_{\Omega^h} \left(\frac{\alert{s_{k+1}^h} - s_k^h}{\Delta t} - \bu|_{s_k^h} \cdot \bn_{s_k^h} - a\right) r^h\,dx + J(\alert{s_{k+1}^h},r^h) = 0$$
for all test functions $r^h$, where
    $$J(s,r) = \sum_{e\in E^h} \int_e \tfrac{1}{2} h_e^2 \left|\bu|_{s_k^h}\right| \jump{\grad s} \jump{\grad r}\,dS$$

    \begin{itemize}
    \item[$\circ$] $s_k^h$ is the FE approximation of surface elevation at time $t_k$
    \item[$\circ$] $E^h$ is the set of interior edges in the triangulation $\Omega^h$
    \item[$\circ$] $\jump{\bv} = \bv_+ \cdot \bn_+ - \bv_- \cdot \bn_-$ is the \alert{jump of a vector field along an edge}
    \item[$\circ$] $h_e$ is the average of the diameters of the two cells which meet $e$
    \item[$\circ$] note surface velocity and slope are evaluated explicitly, at $t_k$
    \item[$\circ$] the system to solve for $\alert{s_{k+1}^h}$ is \alert{linear}
    \end{itemize}
\end{itemize}
\end{frame}


\begin{frame}{technique 1B: CFL determination of time step}

\begin{itemize}
\item our application has an advancing glacier margin, which is not considered in the proof of stability in Tominec et al (2026)

\item note that the edge stabilization technique applies semi-implicitly, with \alert{velocity evaluated explicitly}

\medskip
\item the time step $\Delta t$ must be limited by a CFL\footnote{$=$ Courant-Friedrichs-Lewy} criterion:
    $$\alert{\Delta t = C_{\text{CFL}} \frac{\min h}{\max \big|\bu\big|}}$$

    \begin{itemize}
    \item[$\circ$] margin advance limited to one (base mesh) element per time step
    \item[$\circ$] we get reasonable results with $C_{\text{CFL}}=0.25$ \strut
        \begin{itemize}
        \item Ern \& Guermond (2013) use $C_{\text{CFL}}=0.2$ or $0.25$ with weighted edge stabilization and SSP3, RK4 schemes
        %\item contrast the steady-state analysis in Burman \& Hansbro (2004)
        \end{itemize}
    \end{itemize}
\end{itemize}
\end{frame}


\begin{frame}{technique 2: semi-implicit coupling heuristic in stress balance}

\begin{itemize}
\item also from Tominec et al (2026), but based on L\"ofgren et al (2022)
\item add \alert{extrapolated surface mass balance load} to the Stokes form:
{\small
\begin{align*}
\int_{\Lambda(t_k)} 2\, \nu_\eps(|D\bu|)\, D\bu\,:\,D\bv - p \nabla \cdot \bv - \rho \mathbf{g} \cdot \bv\,dx \qquad& \\
\alert{+ \int_{\Gamma_s(t_k)} \rho g \Delta t \left[\frac{1}{2} (\bu|_{s_k}\cdot \bn_{s_k}) + a\right] (\bv\cdot \bn) \,ds} &= 0
\end{align*}

    \begin{itemize}
    \item[$\circ$] $\Lambda(t_k)$ is the Stokes domain defined by the current surface elevation $s_k$
    \item[$\circ$] $\Gamma_s(t_k)$ is the upper surface of $\Lambda(t_k)$
    \item[$\circ$] $\bn_{s_k}=\left<-\grad s_k,1\right>$ is an outward normal on $\Gamma_s(t_k)$
    \item[$\circ$] $\bn=\bn_{s_k}/|\bn_{s_k}|$ is the outward \emph{unit} normal 
    \end{itemize}
}

\medskip
\item concept: modifying the stress balance stabilizes the coupling
    \begin{itemize}
    \item[$\circ$] Reynold's transport theorem motivates this stabilization
    \item[$\circ$] \dots but the above form is additionally symmetrized
    \item[$\circ$] Tominec et al (2026) \alert{prove} that techniques 1A \& 2 together stabilize the $L^2$ norm of the evolving surface
    \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}{technique 3A: variational inequality for free surface}

\begin{itemize}
\item define the FE \alert{admissible set} $\cK^h = \left\{r^h \in \text{P}_1 \,:\, r^h\ge b^h\right\}$
    \begin{itemize}
    \item[$\circ$] the $P_1$ bed elevation function $b^h$ is time-independent
    \end{itemize}
\item the new surface \alert{$s_{k+1}^h \in \cK^h$} solves this variational inequality weak form:
    $$\int_\Omega \left(\frac{s_{k+1}^h - s_k^h}{\Delta t} - \bu|_{s_k^h} \cdot \bn_{s_k^h} - a\right)\, (r^h-s_{k+1}^h)\,dx + J(s_{k+1}^h,r^h-s_{k+1}^h) \ge 0$$
for all $r^h$ in $\cK^h$

    \begin{itemize}
    \item[$\circ$] includes edge stabilization
    \end{itemize}

\item the computed margin location

is where $s_{k+1}^h$ meets $b^h$

\item we apply the \texttt{vinewtonrsls}

bounds-respecting Newton

solver from PETSc (B 2021b)

\end{itemize}
\end{frame}


\begin{frame}{technique 3B: extruded mesh with variable ice extent}

\begin{itemize}
\item the base mesh covers all of $\Omega$, \alert{including areas with no ice}

\medskip
\item the extruded mesh has a \alert{fixed number \texttt{mz} of levels in the vertical}
    \begin{itemize}
    \item[$\circ$] $\bu,p$ are defined at each node of the extruded mesh
    \item[$\circ$] our meshes and fields have \alert{constant memory footprint} as we time-step
    \end{itemize}

\medskip
\item concept: if $s^h_k=b^h$ at a base mesh node then there is no ice in that column of the extruded mesh
    \begin{itemize}
    \item[$\circ$] note that $s^h_k$ and $b^h$ define the ice domain $\Lambda(t_k)$
    \end{itemize}

\medskip
\item \alert{in columns with no ice we lock $\bu,p$ values to zero}
    \begin{itemize}
    \item[$\circ$] classes \texttt{PinchColumnPressure|Velocity} are sub-classes of \texttt{DirichletBC}; see \texttt{stage4/geometry.py}

\end{itemize}

\begin{frame}[fragile]
\frametitle{technique 3C: pressure elements with interior degrees of freedom}

\begin{itemize}
\item as on previous slide, in columns with no ice we lock $\bu,p$ values to zero
\item however, we do \alert{not} want to lock $p$ to be zero anywhere in the closest-to-margin elements containing ice
    \begin{itemize}
    \item[$\circ$] doing so is an ``accidental'' Dirichlet condition on the pressure
    \end{itemize}
\item thus we prefer elements for pressure which only have interior degrees of freedom, for example:

\item for such elements the moving-margin mass conservation error (later slides) is one-sided
\end{itemize}

        \item not close to Halfar margin location (next slides)  
        \end{itemize}
    \item[$\circ$] but the margin shape is not wildly unstable
    \end{itemize}
\end{itemize}
\end{frame}


\begin{frame}{assessing correctness}

\begin{itemize}
\item[Q.] is our time-dependent solver correct?

\medskip
\item it is not clear how to build a non-trivial moving-margin exact solution of the coupled equations
\item however, such exact solutions exist for \alert{shallow ice approximation (SIA)}
    \begin{itemize}
    \item[$\circ$] both time-dependent (Halfar) and steady-state exact solutions are known
    \end{itemize}
\item SIA exact solutions are \alert{close} to solving the free-surface $+$ Stokes coupled equations when the aspect ratio $\eps = (\text{height}) / (\text{width})$ is small

\bigskip
\item next slide: we compare with the Halfar (1981) exact solution
    \begin{itemize}
    \item[$\circ$] on a \alert{shallow initial dome geometry}, so SIA is reasonably accurate
    \item[$\circ$] initial height 1000 m and width 20 km \quad $\implies$ \quad $\eps = 0.05$
    %\item[$\circ$] Halfar characteristic time $t_0=0.17$ a
    \end{itemize}
\end{itemize}

\begin{frame}{mass conservation errors with moving glacier margins}

\begin{itemize}
\item since $a=0$ for this moving margin solution, mass is exactly conserved in the continuum fluid model

    \begin{itemize}
    \item[$\circ$] curves should be horizontal lines
    \item[$\circ$] higher spatial resolution gives better mass conservation

\section{\hspace{22mm} summary and learning more}


\begin{frame}{summary and outlook}

\begin{itemize}
\item Firedrake makes the solution of the glacer Stokes problem easy
    \begin{itemize}
    \item[$\circ$] the Python codes in this tutorial are short
    \item[$\circ$] UFL (for weak forms) and PETSc (for nonlinear solvers) are key in the stack
    \end{itemize}
\item extruded meshes make sense because glaciers are (generally) thin flows

\bigskip
\item major remaining challenges:
    \begin{enumerate}
    \item make the Stokes solver faster by replacing the direct linear algebra with a \alert{scalable matrix-free solver}
    \item make time-stepping glacier simulations \alert{mass conserving even in free-boundary cases}
    \item make the simulation faster by combining the free-surface equation and the Stokes solver in a \alert{monolithic implicit step}
    \end{enumerate}
\end{itemize}
\end{frame}
